git status (проверить есть ли изменения)
git add .  (добавить ВСЕ измения)
git commit -m 'коммент' (добавить комментарий)
git push -u origin master (не уверен, вернть на сервер)
git clone ссылка (скопировать репозиторий по ссылке)
git init (создание репозитория)
git show (посмотреть последние изменение)
git log (посмотреть все изменения)
git add index.htm (изменить только index.html)
git pull
git remote add origin (добавить связь с внешним репозиторием)
git remote remove origin ссылка (удалить связь с внешним репозиторием)
git remote -v (показывает связи с репозиториями)
